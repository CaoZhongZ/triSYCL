182  /** Transfer the most up-to-date version of the data to the host
183  if the host version is not already up-to-date
184  */
185  void sync_with_host(std::size_t size, void* data) {
186  cl::sycl::context host_context;
187  if (!is_data_up_to_date(host_context) && !fresh_ctx.empty()) {
188  /* We know that the context(s) in \c fresh_ctx hold the most recent
189  version of the buffer
190  */
191  auto fresh_context = *(fresh_ctx.begin());
192  auto fresh_q = fresh_context.get_boost_queue();
193  fresh_q.enqueue_read_buffer(buffer_cache[fresh_context], 0, size, data);
194  fresh_ctx.insert(host_context);
195  }
196  }

199  /** When a transfer is requested this function is called, it will
200  update the state of the buffer according to the context in which
201  the accessor is created and the access mode
202  */
203  void update_buffer_state(const cl::sycl::context& target_ctx,
204  access::mode mode, std::size_t size, void* data) {
205  /* The \c cl_buffer we put in the cache might get accessed again in the
206  future, this means that we have to always to create it in read/write
207  mode to be able to write to it if it is accessed through a
208  write accessor in the future
209  */
210  auto constexpr flag = CL_MEM_READ_WRITE;
211 
212  /* The buffer is accessed in read mode, we want to transfer the data only if
213  necessary. We start a transfer if the data on the target context is not
214  up to date and then update the fresh context set.
215  */
216  if (mode == access::mode::read) {
217 
218  if (is_data_up_to_date(target_ctx))
219  // If read mode and the data is up-to-date there is nothing to do
220  return;
221 
222  // The data is not up-to-date, we need a transfer
223  // We also want to be sure that the host holds the most recent data
224  sync_with_host(size, data);
225 
226  if (!target_ctx.is_host()) {
227  // If the target context is a device context
228  if (!is_cached(target_ctx)) {
229  /* If not cached, we create the buffer and copy the data
230  at the same time
231  */
232  create_in_cache(target_ctx, size,
233  (flag | CL_MEM_COPY_HOST_PTR), data);
234  fresh_ctx.insert(target_ctx);
235  return;
236  }
237 
238  /* Else we transfer the data to the existing buffer associated
239  with the target context buffer
240  */
241  auto q = target_ctx.get_boost_queue();
242  q.enqueue_write_buffer(buffer_cache[target_ctx], 0, size, data);
243  fresh_ctx.insert(target_ctx);
244  }
245  return;
246  }
247 
248  /* The buffer might be written to, this means that we have to consider
249  every version of the data obsolete except in the target context
250 
251  We go through the same process as in read mode but in addition
252  we empty the fresh context set and just add the target context
253 
254  If the data is up to date on the target we just have to update
255  the context set and nothing else
256  */
257  if (!is_data_up_to_date(target_ctx)) {
258 
259  if ( mode == access::mode::read_write
260  || mode == access::mode::write
261  || mode == access::mode::atomic) {
262  // If the data is not up-to-date in the target context
263  // We want to host to be up-to-date
264  sync_with_host(size, data);
265 
266  if (!target_ctx.is_host()) {
267  // If the target context is a device context
268  if (!is_cached(target_ctx)) {
269  create_in_cache(target_ctx, size,
270  (flag | CL_MEM_COPY_HOST_PTR), data);
271  }
272  else {
273  // We update the buffer associated with the target context
274  auto q = target_ctx.get_boost_queue();
275  q.enqueue_write_buffer(buffer_cache[target_ctx], 0, size, data);
276  }
277  }
278  }
279 
280  /* When in discard mode we don't need to transfer any data, we just create
281  the \c cl_buffer if it doesn't exist in the cache
282  */
283  if ( mode == access::mode::discard_write
284  || mode == access::mode::discard_read_write) {
285  /* We only need to create the buffer if it doesn't exist
286  but without copying any data because of the discard mode
287  */
288  if (!target_ctx.is_host() && !is_cached(target_ctx)) {
289  // If the context doesn't exist we create it.
290  /* We don't want to transfer any data so we don't
291  add \c CL_MEM_COPY_HOST_PTR
292  */
293  create_in_cache(target_ctx, size, flag, 0);
294  }
295  }
296  }
297  /* Here we are sure that we are in some kind of write mode,
298  we indicate that all contexts except the target context
299  are not up-to-date anymore
300  */
301  fresh_ctx.clear();
302  fresh_ctx.insert(target_ctx);
303  }
