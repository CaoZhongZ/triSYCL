44  /** Represent a cyclic partition.
45 
46  The single array would be partitioned into several small physical
47  memories in this case. These small physical memories can be
48  accessed simultaneously which drive the performance. Each
49  element in the array would be partitioned to each memory in order
50  and cyclically.
51 
52  That is if we have a 4-element array which contains 4 integers
53  0, 1, 2, and 3. If we set factor to 2, and partition
54  dimension to 1 for this cyclic partition array. Then, the
55  contents of this array will be distributed to 2 physical
56  memories: one contains 1, 3 and the other contains 2,4.
57 
58  \param PhyMemNum is the number of physical memories that user wants to
59  have.
60 
61  \param PDim is the dimension that user wants to apply cyclic partition on.
62  If PDim is 0, all dimensions will be partitioned with cyclic order.
63 
64  \todo Deal with multi-dimension array. Now, since we can only deal with
65  1-dim, PDim is set to 1 by default.
66  */
67  template <std::size_t PhyMemNum = 1, std::size_t PDim = 1>
68  struct cyclic {
69  static constexpr auto physical_mem_num = PhyMemNum;
70  static constexpr auto partition_dim = PDim;
71  static constexpr auto partition_type = type::cyclic;
72  };

75  /** Represent a block partition.
76 
77  The single array would be partitioned into several small
78  physical memories and can be accessed simultaneously, too.
79  However, the first physical memory will be filled up first, then
80  the next.
81 
82  That is if we have a 4-element array which contains 4 integers
83  0, 1, 2, and 3. If we set factor to 2, and partition
84  dimension to 1 for this cyclic partition array. Then, the
85  contents of this array will be distributed to 2 physical
86  memories: one contains 1, 2 and the other contains 3,4.
87 
88  \param ElmInEachPhyMem is the number of elements in each physical memory
89  that user wants to have.
90 
91  \param PDim is the dimension that user wants to apply block partition on.
92  If PDim is 0, all dimensions will be partitioned with block order.
93 
94  \todo Deal with multi-dimension array. Now, since we can only deal with
95  1-dim, PDim is set to 1 by default.
96  */

182  /// Construct an array
183  partition_array() {
184  // Add the intrinsic according expressing to the target compiler the
185  // partitioning to use
186  if constexpr (partition_type == partition::type::cyclic)
187  _ssdm_SpecArrayPartition(&(*this)[0], PartitionType::partition_dim,
188  "CYCLIC", PartitionType::physical_mem_num, "");
189  if constexpr (partition_type == partition::type::block)
190  _ssdm_SpecArrayPartition(&(*this)[0], PartitionType::partition_dim,
191  "BLOCK", PartitionType::ele_in_each_physical_mem,
192  "");
193  if constexpr (partition_type == partition::type::complete)
194  _ssdm_SpecArrayPartition(&(*this)[0], PartitionType::partition_dim,
195  "COMPLETE", 0, "");
196  }
