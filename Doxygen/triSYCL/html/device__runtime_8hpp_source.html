49 /// SYCL accessor seen from a device perspective
50 template <typename Accessor>
51 class accessor :
52  public detail::container_element_aspect<typename Accessor::value_type> {
53 
54  /** The pointer to the data
55 
56  For now none of the following works yet for various reasons:
57  \code
58  trisycl::global_ptr<typename Accessor::value_type> buffer;
59  trisycl::global<typename Accessor::value_type *> buffer;
60  __global typename Accessor::pointer buffer;
61  \endcode
62 
63  So use a more trivial approach
64  */
65  TRISYCL_GLOBAL_AS typename Accessor::value_type *buffer;
66 
67  /** The size of the accessor
68 
69  The compiler should optimize it out if it is unused */
70  std::size_t size;
71 
72 public:
73 
74  /// Capture the minimal accessor features
75  accessor(Accessor &a) :
76  /* Initialize the buffer with actually a \c cl_mem since it will be
77  overridden with the kernel argument by the serializer at
78  run-time, but at least from the kernel point of view it has to
79  look like an \c Accessor::value_type * for the kernel
80  outlining */
81  buffer {
82  reinterpret_cast<TRISYCL_GLOBAL_AS typename Accessor::value_type *>
83  (a.implementation->get_order())
84  },
85  size { a.get_count() }
86  {
87 #ifndef TRISYCL_DEVICE
88  /* Register the buffer address to be updated with the final
89  version before the kernel set arg */
90  TRISYCL_DUMP_T("drt::accessor(Accessor &a) : &a = "
91  << (void *) &a << '\n'
92  << "\t&buffer = " << (void *) &buffer
93  << ", Order = " << a.implementation->get_order());
94 #endif
95  }
96 
97 
98  /// use the accessor with integers à la []
99  auto &operator[](std::size_t index) const noexcept {
100  return buffer[index];
101  }
