164  /** Register the accessor once a \c std::shared_ptr is created on it
165 
166  This is to be called from outside once the object is created. It
167  has been tried directly inside the contructor, but calling \c
168  shared_from_this() from the constructor dead-lock with
169  libstdc++6
170 
171  \todo Double-check with the C++ committee on this issue.
172  */
173  void register_accessor() {
174  if (!task->get_queue()->is_host()) {
175  // To keep alive this accessor in the following lambdas
176  auto acc = this->shared_from_this();
177  // Attach the accessor to the task and get its order
178  set_order(task->register_accessor(acc));
179 #ifdef TRISYCL_OPENCL
180  /* Before running the kernel, make sure the cl_mem behind this
181  accessor is up-to-date on the device if needed and pass it to
182  the kernel */
183  task->add_prelude([=] {
184  acc->copy_in_cl_buffer();
185  });
186  // After running the kernel, deal with some copy-back if needed
187  task->add_postlude([=] {
188  /* Even if this function does nothing, it is required to
189  have the capture of acc to keep the accessor alive across
190  the kernel execution up to the execution postlude */
191  acc->copy_back_cl_buffer();
192  });
193 #endif
194  }
195  }
