163  /** Register the accessor once a \c std::shared_ptr is created on it
164 
165  This is to be called from outside once the object is created. It
166  has been tried directly inside the contructor, but calling \c
167  shared_from_this() from the constructor dead-lock with
168  libstdc++6
169 
170  \todo Double-check with the C++ committee on this issue.
171  */
172  void register_accessor() {
173 #ifdef TRISYCL_OPENCL
174  if (!task->get_queue()->is_host()) {
175  // To keep alive this accessor in the following lambdas
176  auto acc = this->shared_from_this();
177  /* Before running the kernel, make sure the cl_mem behind this
178  accessor is up-to-date on the device if needed and pass it to
179  the kernel */
180  task->add_prelude([=] {
181  acc->copy_in_cl_buffer();
182  });
183  // After running the kernel, deal with some copy-back if needed
184  task->add_postlude([=] {
185  /* Even if this function does nothing, it is required to
186  have the capture of acc to keep the accessor alive across
187  the kernel execution up to the execution postlude */
188  acc->copy_back_cl_buffer();
189  });
190  }
191 #endif
