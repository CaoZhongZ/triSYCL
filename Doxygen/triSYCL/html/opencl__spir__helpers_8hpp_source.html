33 // Fills the small_array_123 of the type passed to it up to the dimensionality
34 // of the type T (T generally intended to be a sycl id or range, but can be
35 // anything with a dimensionality variable defined and a subscript operator)
36 template <typename T, typename F>
37 auto make_array_with_func(F unary_func) {
38  T id_or_range {};
39 
40  for (size_t i = 0; i < id_or_range.dimensionality; ++i)
41  id_or_range[i] = unary_func(i);
42 
43  return id_or_range;
44 }

70 // Uses a passed in index or range to identify the required index or range type
71 // to generate
72 template <int Dimensions, typename Index_T>
73 auto create_parallel_for_arg(Index_T index_or_range) {
74  if constexpr (std::is_same_v<Index_T, id<Dimensions>>) {
75  index_or_range = make_array_with_func<Index_T>(get_global_id);
76  } else if constexpr (std::is_same_v<Index_T, item<Dimensions>>) {
77  // Item's do not contain local information in parallel_for
78  index_or_range = Index_T {
79  make_array_with_func<range<Dimensions>>(get_global_size),
80  make_array_with_func<id<Dimensions>>(get_global_id),
81  make_array_with_func<id<Dimensions>>(get_global_offset)
82  };
83  } else {
84  static_assert(std::is_same_v<Index_T, item<Dimensions>> ||
85  std::is_same_v<Index_T, id<Dimensions>>,
86  "undefined Index_T passed to create_parallel_for_arg");
87  }
88 
89  return index_or_range;
90 }
