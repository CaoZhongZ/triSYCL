84  /// Add a new task to the task graph and schedule for execution
85  void schedule(std::function<void(void)> f) {
86  /* To keep a copy of the task shared_ptr after the end of the
87  command group, capture it by copy in the following lambda. This
88  should be easier in C++17 with move semantics on capture
89  */
90  auto task = shared_from_this();
91  auto execution = [=] {
92  // Wait for the required tasks to be ready
93  task->wait_for_producers();
94  task->prelude();
95  TRISYCL_DUMP_T("Execute the kernel");
96  // Execute the kernel
97  f();
98  task->postlude();
99  // Release the buffers that have been written by this task
100  task->release_buffers();
101  // Notify the waiting tasks that we are done
102  task->notify_consumers();
103  // Notify the queue we are done
104  owner_queue->kernel_end();
105  TRISYCL_DUMP_T("Task thread exit");
106  };
107  /* Notify the queue that there is a kernel submitted to the
108  queue. Do not do it in the task contructor so that we can deal
109  with command group without kernel and if we put it inside the
110  thread, the queue may have finished before the thread is
111  scheduled */
112  owner_queue->kernel_start();
113  /* \todo it may be implementable with packaged_task that would
114  deal with exceptions in kernels
115  */
116 #ifndef TRISYCL_NO_ASYNC
117  /* If in asynchronous execution mode, execute the functor in a new
118  thread */
119  std::thread thread(execution);
120  TRISYCL_DUMP_T("Task thread started");
121  /** Detach the thread since it will synchronize by its own means
122 
123  \todo This is an issue if there is an exception in the kernel
124  */
125  thread.detach();
126 #else
127  // Just a synchronous execution otherwise
128  execution();
129 #endif
130  }

176  /** Register a buffer to this task
177 
178  This is how the dependency graph is incrementally built.
179  */
180  void add_buffer(std::shared_ptr<detail::buffer_base> &buf,
181  bool is_write_mode) {
182  TRISYCL_DUMP_T("Add buffer " << buf << " in task " << this);
183  /* Keep track of the use of the buffer to notify its release at
184  the end of the execution */
185  buffers_in_use.push_back(buf);
186  // To be sure the buffer does not disappear before the kernel can run
187  buf->use();
188 
189  std::shared_ptr<detail::task> latest_producer;
190  if (is_write_mode) {
191  /* Set this task as the latest producer of the buffer so that
192  another kernel may wait on this task */
193  latest_producer = buf->set_latest_producer(shared_from_this());
194  }
195  else
196  latest_producer = buf->get_latest_producer();
197 
198  /* If the buffer is to be produced by a task, add the task in the
199  producer list to wait on it before running the task core
200 
201  If a buffer is accessed first in write mode and then in read mode,
202  the task will add itself as a producer and will wait for itself
203  when calling \c wait_for_producers, we avoid this by checking that
204  \c latest_producer is not \c this
205  */
206  if (latest_producer && latest_producer != shared_from_this())
207  producer_tasks.push_back(latest_producer);
208  }
