327  /** Reserve some part of the pipe for reading
328 
329  \param[in] s is the number of element to reserve
330 
331  \param[out] rid is an iterator to a description of the
332  reservation that has been done if successful
333 
334  \param[in] blocking specify if the call wait for the operation
335  to succeed
336 
337  \return true if the reservation was successful
338  */
339  bool reserve_read(std::size_t s,
340  rid_iterator &rid,
341  bool blocking = false) {
342  // Lock the pipe to avoid being disturbed
343  std::unique_lock<std::mutex> ul { cb_mutex };
344 
345  TRISYCL_DUMP_T("Before read reservation cb.size() = " << cb.size()
346  << " size() = " << size());
347  if (s == 0)
348  // Empty reservation requested, so nothing to do
349  return false;
350 
351  if (blocking)
352  /* If in blocking mode, wait for enough elements to read in the
353  pipe for the reservation. This condition can change when a
354  write is done */
355  write_done.wait(ul, [&] { return s <= size(); });
356  else if (s > size())
357  // Not enough elements to read in the pipe for the reservation
358  return false;
359 
360  // Compute the location of the first element of the reservation
361  auto first = cb.begin() + read_reserved_frozen;
362  // Increment the number of frozen elements
363  read_reserved_frozen += s;
364  /* Add a description of the reservation at the end of the
365  reservation queue */
366  r_rid_q.emplace_back(first, s);
367  // Return the iterator to the last reservation descriptor
368  rid = r_rid_q.end() - 1;
369  TRISYCL_DUMP_T("After reservation cb.size() = " << cb.size()
370  << " size() = " << size());
371  return true;
372  }

387  bool reserve_write(std::size_t s,
388  rid_iterator &rid,
389  bool blocking = false) {
390  // Lock the pipe to avoid being disturbed
391  std::unique_lock<std::mutex> ul { cb_mutex };
392 
393  TRISYCL_DUMP_T("Before write reservation cb.size() = " << cb.size()
394  << " size() = " << size());
395  if (s == 0)
396  // Empty reservation requested, so nothing to do
397  return false;
398 
399  if (blocking)
400  /* If in blocking mode, wait for enough room in the pipe, that
401  may be changed when a read is done. Do not use a difference
402  here because it is only about unsigned values */
403  read_done.wait(ul, [&] { return cb.size() + s <= capacity(); });
404  else if (cb.size() + s > capacity())
405  // Not enough room in the pipe for the reservation
406  return false;
407 
408  /* If there is enough room in the pipe, just create default values
409  in it to do the reservation */
410  for (std::size_t i = 0; i != s; ++i)
411  cb.push_back();
412  /* Compute the location of the first element a posteriori since it
413  may not exist a priori if cb was empty before */
414  auto first = cb.end() - s;
415  /* Add a description of the reservation at the end of the
416  reservation queue */
417  w_rid_q.emplace_back(first, s);
418  // Return the iterator to the last reservation descriptor
419  rid = w_rid_q.end() - 1;
420  TRISYCL_DUMP_T("After reservation cb.size() = " << cb.size()
421  << " size() = " << size());
422  return true;

426  /** Process the read reservations that are ready to be released in the
427  reservation queue
428  */
429  void move_read_reservation_forward() {
430  // Lock the pipe to avoid nuisance
431  std::unique_lock<std::mutex> lock { cb_mutex };
432 
433  for (;;) {
434  if (r_rid_q.empty())
435  // No pending reservation, so nothing to do
436  break;
437  if (!r_rid_q.front().ready)
438  /* If the first reservation is not ready to be released, stop
439  because it is blocking all the following in the queue
440  anyway */
441  break;
442  // Remove the reservation to be released from the queue
443  r_rid_q.pop_front();
444  std::size_t n_to_pop;
445  if (r_rid_q.empty())
446  // If it was the last one, remove all the reservation
447  n_to_pop = read_reserved_frozen;
448  else
449  // Else remove everything up to the next reservation
450  n_to_pop = r_rid_q.front().start - cb.begin();
451  // No longer take into account these reserved slots
452  read_reserved_frozen -= n_to_pop;
453  // Release the elements from the FIFO
454  while (n_to_pop--)
455  cb.pop_front();
456  // Notify the clients waiting for some room to write in the pipe
457  read_done.notify_all();
458  /* ...and process the next reservation to see if it is ready to
459  be released too */
460  }
461  }
