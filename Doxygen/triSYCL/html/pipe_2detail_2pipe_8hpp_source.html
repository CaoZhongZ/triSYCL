329  /** Reserve some part of the pipe for reading
330 
331  \param[in] s is the number of element to reserve
332 
333  \param[out] rid is an iterator to a description of the
334  reservation that has been done if successful
335 
336  \param[in] blocking specify if the call wait for the operation
337  to succeed
338 
339  \return true if the reservation was successful
340  */
341  bool reserve_read(std::size_t s,
342  rid_iterator &rid,
343  bool blocking = false) {
344  // Lock the pipe to avoid being disturbed
345  std::unique_lock<std::mutex> ul { cb_mutex };
346 
347  TRISYCL_DUMP_T("Before read reservation cb.size() = " << cb.size()
348  << " size() = " << size());
349  if (s == 0)
350  // Empty reservation requested, so nothing to do
351  return false;
352 
353  if (blocking)
354  /* If in blocking mode, wait for enough elements to read in the
355  pipe for the reservation. This condition can change when a
356  write is done */
357  write_done.wait(ul, [&] { return s <= size(); });
358  else if (s > size())
359  // Not enough elements to read in the pipe for the reservation
360  return false;
361 
362  // Compute the location of the first element of the reservation
363  auto first = cb.begin() + read_reserved_frozen;
364  // Increment the number of frozen elements
365  read_reserved_frozen += s;
366  /* Add a description of the reservation at the end of the
367  reservation queue */
368  r_rid_q.emplace_back(first, s);
369  // Return the iterator to the last reservation descriptor
370  rid = r_rid_q.end() - 1;
371  TRISYCL_DUMP_T("After reservation cb.size() = " << cb.size()
372  << " size() = " << size());
373  return true;
374  }

389  bool reserve_write(std::size_t s,
390  rid_iterator &rid,
391  bool blocking = false) {
392  // Lock the pipe to avoid being disturbed
393  std::unique_lock<std::mutex> ul { cb_mutex };
394 
395  TRISYCL_DUMP_T("Before write reservation cb.size() = " << cb.size()
396  << " size() = " << size());
397  if (s == 0)
398  // Empty reservation requested, so nothing to do
399  return false;
400 
401  if (blocking)
402  /* If in blocking mode, wait for enough room in the pipe, that
403  may be changed when a read is done. Do not use a difference
404  here because it is only about unsigned values */
405  read_done.wait(ul, [&] { return cb.size() + s <= capacity(); });
406  else if (cb.size() + s > capacity())
407  // Not enough room in the pipe for the reservation
408  return false;
409 
410  /* If there is enough room in the pipe, just create default values
411  in it to do the reservation */
412  for (std::size_t i = 0; i != s; ++i)
413  cb.push_back();
414  /* Compute the location of the first element a posteriori since it
415  may not exist a priori if cb was empty before */
416  auto first = cb.end() - s;
417  /* Add a description of the reservation at the end of the
418  reservation queue */
419  w_rid_q.emplace_back(first, s);
420  // Return the iterator to the last reservation descriptor
421  rid = w_rid_q.end() - 1;
422  TRISYCL_DUMP_T("After reservation cb.size() = " << cb.size()
423  << " size() = " << size());
424  return true;

428  /** Process the read reservations that are ready to be released in the
429  reservation queue
430  */
431  void move_read_reservation_forward() {
432  // Lock the pipe to avoid nuisance
433  std::unique_lock<std::mutex> lock { cb_mutex };
434 
435  for (;;) {
436  if (r_rid_q.empty())
437  // No pending reservation, so nothing to do
438  break;
439  if (!r_rid_q.front().ready)
440  /* If the first reservation is not ready to be released, stop
441  because it is blocking all the following in the queue
442  anyway */
443  break;
444  // Remove the reservation to be released from the queue
445  r_rid_q.pop_front();
446  std::size_t n_to_pop;
447  if (r_rid_q.empty())
448  // If it was the last one, remove all the reservation
449  n_to_pop = read_reserved_frozen;
450  else
451  // Else remove everything up to the next reservation
452  n_to_pop = r_rid_q.front().start - cb.begin();
453  // No longer take into account these reserved slots
454  read_reserved_frozen -= n_to_pop;
455  // Release the elements from the FIFO
456  while (n_to_pop--)
457  cb.pop_front();
458  // Notify the clients waiting for some room to write in the pipe
459  read_done.notify_all();
460  /* ...and process the next reservation to see if it is ready to
461  be released too */
462  }
463  }
