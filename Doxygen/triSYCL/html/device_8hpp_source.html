92  auto devices = device::get_devices();
93  if (devices.empty())
94  // \todo Put a SYCL exception

96 
97  /* Find the device with the best score according to the given
98  device_selector */
99  auto max = std::max_element(devices.cbegin(), devices.cend(),
100  [&] (const device &d1, const device &d2) {
101  return ds(d1) < ds(d2);
102  });
103  if (ds(*max) < 0)
104  // \todo Put a SYCL exception
105  throw std::domain_error("No device selected because no positive "
106  "device_selector score found");

108  // Create the current device as a shared copy of the selected one
109  implementation = max->implementation;
110  }
