38 /** A recursive multi-dimensional iterator that ends up calling f
39 
40  The iteration order may be changed later.
41 
42  Since partial specialization of function template is not possible in
43  C++14, use a class template instead with everything in the
44  constructor.
45 */
46 template <std::size_t level, typename Range, typename ParallelForFunctor, typename Id>
47 struct parallel_for_iterate {
48  parallel_for_iterate(Range r, ParallelForFunctor &f, Id &index) {
49  for (boost::multi_array_types::index _sycl_index = 0,
50  _sycl_end = r[Range::dimensionality - level];
51  _sycl_index < _sycl_end;
52  _sycl_index++) {
53  // Set the current value of the index for this dimension
54  index[Range::dimensionality - level] = _sycl_index;
55  // Iterate further on lower dimensions
56  parallel_for_iterate<level - 1,
57  Range,
58  ParallelForFunctor,
59  Id> { r, f, index };
60  }
61  }
62 };

65 /** A top-level recursive multi-dimensional iterator variant using OpenMP
66 
67  Only the top-level loop uses OpenMP and goes on with the normal
68  recursive multi-dimensional.
69 */
70 template <std::size_t level,
71  typename Range,
72  typename ParallelForFunctor,
73  typename Id>
74 struct parallel_OpenMP_for_iterate {
75  parallel_OpenMP_for_iterate(Range r, ParallelForFunctor &f) {
76  // Create the OpenMP threads before the for-loop to avoid creating an
77  // index in each iteration
78 #pragma omp parallel
79  {
80  // Allocate an OpenMP thread-local index
81  Id index;
82  // Make a simple loop end condition for OpenMP
83  boost::multi_array_types::index _sycl_end =
84  r[Range::dimensionality - level];
85  /* Distribute the iterations on the OpenMP threads. Some OpenMP
86  "collapse" could be useful for small iteration space, but it
87  would need some template specialization to have real contiguous
88  loop nests */
89 #pragma omp for
90  for (boost::multi_array_types::index _sycl_index = 0;
91  _sycl_index < _sycl_end;
92  _sycl_index++) {
93  // Set the current value of the index for this dimension
94  index[Range::dimensionality - level] = _sycl_index;
95  // Iterate further on lower dimensions
96  parallel_for_iterate<level - 1,
97  Range,
98  ParallelForFunctor,
99  Id> { r, f, index };
100  }
101  }
102  }
103 };
