28 /** Helper to construct an array from initializer elements provided as a
29  tuple
30 
31  The trick is to get the std::index_sequence<> that represent 0,
32  1,..., dimension-1 as a variadic template pack Is that we can
33  iterate on, in this function.
34 */
35 template <typename V, typename Tuple, size_t... Is>
36 std::array<typename V::element_type, V::dimension>
37 tuple_to_array_iterate(Tuple t, std::index_sequence<Is...>) {
38  /* The effect is like a static for-loop with Is counting from 0 to
39  dimension-1 and thus constructing a uniform initialization { }
40  construction from each tuple element:
41  { std::get<0>(t), std::get<1>(t), ..., std::get<dimension-1>(t) }
42 
43  The static cast is here to avoid the warning when there is a loss
44  of precision, for example when initializing an int from a float.
45  */
46  return { { static_cast<typename V::element_type>(std::get<Is>(t))...} };
47 }

82  /** Construct a tuple from a value
83 
84  \param value is used to initialize each tuple element
85 
86  \param size is the number of elements of the tuple to be generated
87 
88  The trick is to get the std::index_sequence<> that represent 0,
89  1,..., dimension-1 as a variadic template pack Is that we can
90  iterate on, in this function.
91  */
92  template <typename Value, size_t... Is>
93  static auto fill_tuple(Value e, std::index_sequence<Is...>) {
94  /* The effect is like a static for-loop with Is counting from 0 to
95  dimension-1 and thus replicating the pattern to have
96  make_tuple( (0, e), (1, e), ... (n - 1, e) )
97 
98  Since the "," operator is just here to throw away the Is value
99  (which is needed for the pack expansion...), at the end this is
100  equivalent to:
101  make_tuple( e, e, ..., e )
102  */
103  return std::make_tuple(((void)Is, e)...);
104  }
