65 /** Define a multi-dimensional index, used for example to locate a work
66  item or a buffer element
67 
68  Unfortunately, even if std::array is an aggregate class allowing
69  native list initialization, it is no longer an aggregate if we derive
70  from an aggregate. Thus we have to redeclare the constructors.
71 
72  \param BasicType is the type element, such as int
73 
74  \param Dims is the dimension number, typically between 1 and 3
75 
76  \param FinalType is the final type, such as range<> or id<>, so that
77  boost::operator can return the right type
78 
79  \param EnableArgsConstructor adds a constructors from Dims variadic
80  elements when true. It is false by default.
81 
82  std::array<> provides the collection concept, with .size(), == and !=
83  too.
84 
85  Can't use boost::shiftable, as it doesn't define the U, T overload
86  outside the class, but when that is added, it gets ambiguity problems.
87 */
88 template <typename BasicType,
89  typename FinalType,
90  std::size_t Dims,
91  bool EnableArgsConstructor = false>
92 struct small_array : std::array<BasicType, Dims>,
93  // To have all the usual arithmetic operations on this type and bitwise
94  // operations. Note these opeation classes are chained via:
95  // https://www.boost.org/doc/libs/1_69_0/libs/utility/operators.htm#chaining
96  // without the chaining the empty base class optimization will break and the
97  // class will bloat other types like vec.hpp
98  boost::euclidean_ring_operators<FinalType,
99  boost::euclidean_ring_operators<FinalType, BasicType,
100  boost::bitwise<FinalType,
101  boost::bitwise<FinalType, BasicType>>>>,
102  // Already provided by array<> lexicographically:
103  // boost::equality_comparable<FinalType>,
104  // boost::less_than_comparable<FinalType>,
105  // Add a display() method
