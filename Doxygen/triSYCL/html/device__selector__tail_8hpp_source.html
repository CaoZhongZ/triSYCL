25 inline device device_selector::select_device(vector_class<platform> platforms) const {
26  int best_id = -1;
27  int best_score = -1;
28  int i = 0;
29  device best_device;
30 
31  for (auto &platform : platforms) {
32  for (auto &dev : platform.get_devices()) {
33  int score = operator()(dev);
34  if (score > best_score) {
35  best_id = 1;
36  best_device = dev;
37  best_score = score;
38  }
39  ++i;
40  }
41  }
42  if (best_id == -1)
43  throw ::trisycl::runtime_error("Could not find device");
44 
45  return best_device;
46 }
